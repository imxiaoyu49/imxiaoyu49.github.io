<script>
let score = 0;

function nextQ(n) {
    const radios = document.getElementsByName("q" + n);
    for (let r of radios) {
        if (r.checked) {
            score += parseInt(r.value);
            break;
        }
    }
    document.getElementById("q" + n).classList.remove("active");
    const next = document.getElementById("q" + (n + 1));
    if (next) next.classList.add("active");
}

function triggerPopup() {
    nextQ(4);
    alert("为什么你不想睡");
    document.getElementById("subtitle").innerText = "我知道你不睡是因为我…";
}

function openMany() {
    for(let i=0; i<3; i++) {
        window.open(location.href, "_blank");
    }
    nextQ(8);
    document.getElementById("subtitle").innerText = "我把你关在这里你逃不掉…";
}

function finish() {
    nextQ(12);
    document.getElementById("result").classList.add("active");
    let text;
    if (score < 18) text = "我仍然尊重你，但我知道你的弱点。";
    else if (score < 28) text = "你在抗拒，但我在拉近你。";
    else text = "你在问我为什么停不下来看你。";
    document.getElementById("final").innerText = text;
    document.getElementById("subtitle").innerText = "结束仅代表开始 小鱼";
}

window.onbeforeunload = function() {
    return "你真的离开我吗";
};

</script>
